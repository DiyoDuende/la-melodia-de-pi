// ============================================================ // 🎼 GENERADOR DE PI // ============================================================ function* generarPi() { let q = 1n, r = 0n, t = 1n, k = 1n, n = 3n, l = 3n; while (true) { if (4n*q + r - t < n*t) { yield Number(n); let nr = 10n*(r - n*t); n = (10n*(3n*q + r))/t - 10n*n; q = 10n*q; r = nr; } else { let nr = (2n*q + r)*l; let nn = (q*(7n*k) + 2n + r*l)/(t*l); q = q*k; t = t*l; l = l + 2n; k = k + 1n; n = nn; r = nr; } } } // ============================================================ // 🎼 NOTAS // ============================================================ const NOTAS = { '0':'Mi⁸','1':'Do','2':'Re','3':'Mi','4':'Fa', '5':'Sol','6':'La','7':'Si','8':'Do⁸','9':'Re⁸' }; // ============================================================ // GENERAR NOTAS // ============================================================ function generarNotas(inicio, cantidad) { const gen = generarPi(); const cache = []; function get(i) { while (cache.length <= i) { cache.push(gen.next().value.toString()); } return cache[i]; } const lista = []; for (let i = 0; i < cantidad; i++) { const idx = inicio + i; const d = get(idx); lista.push({ segundo: i, digito: d, nota: NOTAS[d] }); } return lista; } // ============================================================ // PDF // ============================================================ function generarPDF(interprete) { const { jsPDF } = window.jspdf; const doc = new jsPDF(); const inicio = (interprete.posicion - 1) * 300; const notas = generarNotas(inicio, 300); doc.setFontSize(16); doc.text("La melodía de π", 20, 20); doc.setFontSize(10); doc.text(`Código: ${interprete.codigo}`, 20, 30); doc.text(`Turno: ${new Date(interprete.turnoUTC).toUTCString()}`, 20, 36); doc.text(`Duración: 5 minutos (300 notas)`, 20, 42); let y = 55; notas.forEach(n => { doc.text(`${n.segundo} | ${n.digito} | ${n.nota}`, 20, y); y += 6; if (y > 280) { doc.addPage(); y = 20; } }); doc.save(`pi-${interprete.codigo}.pdf`); }
